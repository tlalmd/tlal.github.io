Cholesterol Trends Across Race/Ethnicity Groups

ui <- dashboardPage(
  dashboardHeader(title = "Cholesterol Trends Across Race/Ethnicity Groups"),
  dashboardSidebar(
    sidebarMenu(
      menuItem("Overview", tabName = "overview", icon = icon("info")),
      menuItem("Data Exploration", tabName = "exploration", icon = icon("chart-bar")),
      menuItem("Model Diagnostics", tabName = "diagnostics", icon = icon("tools")),
      menuItem("Analysis of Predictors", tabName = "predictors", icon = icon("sliders-h"))
    )
  ),
  dashboardBody(
    tabItems(
      tabItem(
        tabName = "overview",
        h2("Cholesterol Study Overview"),
        p("This dashboard explores the relationship between race/ethnicity and total cholesterol levels."),
        DTOutput("summary_table")
      ),
      tabItem(
        tabName = "exploration",
        h2("Data Exploration"),
        fluidRow(
          box(
            title = "Missingness Before Imputation",
            plotOutput("missingness_plot"),
            width = 6
          ),
          box(
            title = "Cholesterol Distribution",
            plotOutput("cholesterol_histogram"),
            width = 6
          )
        )
      ),
      tabItem(
        tabName = "diagnostics",
        h2("Model Diagnostics"),
        fluidRow(
          box(title = "Residuals vs Fitted (Full Model)", plotOutput("residuals_plot"), width = 6)
        ),
        verbatimTextOutput("model_performance")
      ),
      tabItem(
        tabName = "predictors",
        h2("Analysis of Predictors"),
        plotOutput("race_ethnicity_plot"),
        DTOutput("regression_table")
      )
    )
  )
)

server <- function(input, output) {
  output$summary_table <- renderDT({
    study2_data %>%
      group_by(Race_Ethnicity) %>%
      summarise(
        Mean_Cholesterol = mean(LBXTC, na.rm = TRUE),
        SD_Cholesterol = sd(LBXTC, na.rm = TRUE),
        N = n()
      ) %>%
      datatable()
  })
  output$missingness_plot <- renderPlot({
    missing_summary <- study2_data %>%
      summarise(across(everything(), ~ sum(is.na(.)))) %>%
      pivot_longer(cols = everything(), names_to = "variable", values_to = "n_miss")
    ggplot(missing_summary, aes(x = variable, y = n_miss)) +
      geom_col(fill = "coral") +
      labs(title = "Missing Data by Variable", x = "Variable", y = "# Missing") +
    theme(axis.text.x = element_text(angle = 45, hjust = 1, vjust = 1))
  })
  output$cholesterol_histogram <- renderPlot({
    ggplot(study2_data_imputed, aes(x = LBXTC)) +
      geom_histogram(bins = 30, fill = "coral", color = "darkred") +
      labs(title = "Distribution of Total Cholesterol", x = "Cholesterol (mg/dL)", y = "Frequency")
  })
  output$residuals_plot <- renderPlot({
    ggplot(full_model, aes(x = .fitted, y = .resid)) +
      geom_point(alpha = 0.5) +
      geom_hline(yintercept = 0, linetype = "dashed", color = "darkred") +
      labs(title = "Residuals vs Fitted (Full Model)", x = "Fitted Values", y = "Residuals")
  })
  output$model_performance <- renderPrint({
    data.frame(
      RMSE = 181.7983,
      MAE = 177.1574
    )
  })
  output$race_ethnicity_plot <- renderPlot({
    ggplot(study2_data_imputed, aes(x = Race_Ethnicity, y = LBXTC)) +
      geom_boxplot(fill = "coral", color = "darkred") +
      labs(title = "Cholesterol Levels by Race/Ethnicity", x = "Race/Ethnicity", y = "Total Cholesterol (mg/dL)")
  })
  output$regression_table <- renderDT({
    tidy_full <- tidy(full_model, conf.int = TRUE, conf.level = 0.90)
    tidy_full %>%
      select(term, estimate, std.error, p.value, conf.low, conf.high) %>%
      datatable() %>%
      formatRound(columns = c("estimate", "std.error", "p.value", "conf.low", "conf.high"), digits = 4)
  })
}

shinyApp(ui = ui, server = server)
